1. ##### this指向

   ```js
   var length = 10;
   
   function fn(){
       console.log(this.length);
   }
   
   var obj = {
       length:5,
       method:function(a){
           a();              //10
           console.log(arguments[0]);  //函数
           arguments[0](); //3
           a.call(obj,12); //5
       }
   }
   
   obj.method(fn,1,2);
   ```
