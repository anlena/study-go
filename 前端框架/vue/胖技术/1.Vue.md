2. 关于一个排序

   ```html
   <ul>
   	<li v-for="item in sortItems">{{item}}</li>
   </ul>
   ```

   ```js
   var app = new Vue({
     el: '#app',
     data: {
     	items: [20, 23, 18, 65, 32, 19, 5, 56, 41]
     },
     computed: {
       sortItems: function() {
       return this.items.sort(sortNumber);
     }
   }
   });

   function sortNumber(a, b) {
   	return a - b;
   }
   ```
